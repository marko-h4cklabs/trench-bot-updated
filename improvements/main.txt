// Ensure transaction monitor starts only once

var transactionMonitorMutex = &sync.Mutex{}

func main() {
	// âœ… Recover from unexpected panics
	defer func() {
		if r := recover(); r != nil {
			log.Fatalf("âŒ Program crashed with panic: %v", r)
		}
	}()

	// âœ… Set Gin mode to release for production
	gin.SetMode(gin.ReleaseMode)
	log.Println("âœ… Gin mode set to release.")

	// âœ… Load environment variables
	envPath := filepath.Join("/home/markosrnec5/ca-scraper/agent/.env")
	if err := godotenv.Load(envPath); err != nil {
		log.Println("âš ï¸ Warning: .env file not found, using system environment variables.")
	} else {
		log.Println("âœ… Environment variables loaded successfully.")
	}

	// âœ… Load configuration
	cfg, err := config.LoadConfig("agent/config.yaml")
	if err != nil {
		log.Fatalf("âŒ Failed to load configuration: %v", err)
	}
	log.Println("âœ… Loaded configuration successfully.")

	// âœ… Save configuration globally
	config.SetGlobalConfig(cfg)

	// âœ… Initialize Logger
	appLogger, err := logger.NewLogger(cfg.App.Environment, true)
	if err != nil {
		log.Fatalf("âŒ Failed to initialize logger: %v", err)
	}
	log.Println("âœ… Logger initialized successfully.")

	// âœ… Initialize Telegram Bot
	log.Println("ðŸš€ Initializing Telegram Bot...")
	if err := bot.InitializeBot(); err != nil {
		log.Fatalf("âŒ Failed to initialize Telegram Bot: %v", err)
	}
	log.Println("âœ… Telegram Bot initialized successfully.")

	// âœ… Fetch SOL price at startup
	if price, err := services.FetchSOLPrice(); err != nil {
		log.Printf("âš ï¸ Warning: Failed to fetch SOL price: %v", err)
	} else {
		log.Printf("âœ… SOL Price at startup: $%.2f", price)
	}

	// âœ… Initialize Solana settings
	log.Println("ðŸš€ Initializing Solana settings...")
	services.InitializeSolana()
	log.Println("âœ… Solana settings initialized.")

	// âœ… Determine application port
	port := cfg.App.Port
	if port == "" {
		port = "5555"
	}
	log.Printf("âœ… Application will run on port: %s", port)

	// âœ… Create Gin router
	log.Println("ðŸš€ Creating Gin router...")
	router := gin.Default()

	// âœ… Enable CORS for frontend requests
	router.Use(cors.Default())

	// âœ… Set trusted proxies for security
	log.Println("âœ… Setting trusted proxies to nil.")
	router.SetTrustedProxies(nil)

	// âœ… Register API routes
	log.Println("ðŸš€ Registering API routes...")
	handlers.RegisterRoutes(router, appLogger.ZapLogger, appLogger)

	// âœ… Start Gin Server in a Goroutine
	go func() {
		log.Printf("ðŸš€ Server is now running and listening on: http://localhost:%s", port)
		if err := router.Run(":" + port); err != nil {
			log.Fatalf("âŒ Could not start server: %v", err)
		}
	}()

	// âœ… Run Startup Health Checks
	go RunStartupTests()

	// âœ… Keep the program alive (prevents main from exiting)
	select {}
}

// âœ… Run Startup Tests
func RunStartupTests() {
	log.Println("â³ Waiting for the server to start before running tests...")
	time.Sleep(5 * time.Second)

	log.Println("ðŸš€ Running Startup Health Checks...")

	allTestsPassed := true

	endpoints := []struct {
		path   string
		method string
	}{
		{"/api/v1/health", "GET"},
		{"/api/v1/dexscreener", "GET"},
	}

	for _, ep := range endpoints {
		log.Printf("ðŸ” Running Test: %s [%s]", ep.method, ep.path)
		if !testLocalAPI(ep.path, ep.method) {
			allTestsPassed = false
			log.Printf("âŒ Test Failed: %s [%s]", ep.method, ep.path)
		} else {
			log.Printf("âœ… Test Passed: %s [%s]", ep.method, ep.path)
		}
	}

	if allTestsPassed {
		log.Println("âœ… All Startup Tests Passed Successfully! ðŸŽ‰")
	} else {
		log.Println("âŒ Some Tests Failed. Check logs for details.")
	}
}

// âœ… Test Local API Endpoints
func testLocalAPI(endpoint string, method string) bool {
	url := fmt.Sprintf("http://localhost:5555%s", endpoint)
	resp, err := http.Get(url)

	if err != nil {
		log.Printf("âŒ Failed to connect to %s: %v", endpoint, err)
		return false
	}
	defer resp.Body.Close()

	return resp.StatusCode >= 200 && resp.StatusCode < 300
}
